## Token Generation

```typescript
import jwt from 'jsonwebtoken'

const JWT_SECRET = process.env.JWT_SECRET!
const JWT_EXPIRES_IN = '15m'
const REFRESH_EXPIRES_IN = '7d'

export function generateTokens(userId: string) {
  const accessToken = jwt.sign({ userId }, JWT_SECRET, {
    expiresIn: JWT_EXPIRES_IN,
  })

  const refreshToken = jwt.sign({ userId, type: 'refresh' }, JWT_SECRET, {
    expiresIn: REFRESH_EXPIRES_IN,
  })

  return { accessToken, refreshToken }
}
```

## Auth Middleware

```typescript
import { Request, Response, NextFunction } from 'express'
import jwt from 'jsonwebtoken'

export function authMiddleware(req: Request, res: Response, next: NextFunction) {
  const header = req.headers.authorization

  if (!header?.startsWith('Bearer ')) {
    return res.status(401).json({ error: 'No token provided' })
  }

  const token = header.split(' ')[1]

  try {
    const payload = jwt.verify(token, JWT_SECRET) as { userId: string }
    req.userId = payload.userId
    next()
  } catch (error) {
    return res.status(401).json({ error: 'Invalid or expired token' })
  }
}
```

## Login Route

```typescript
app.post('/auth/login', async (req, res) => {
  const { email, password } = req.body

  const user = await db.users.findOne({ email })
  if (!user || !(await bcrypt.compare(password, user.passwordHash))) {
    return res.status(401).json({ error: 'Invalid credentials' })
  }

  const tokens = generateTokens(user._id.toString())

  // Store refresh token
  await db.refreshTokens.insertOne({
    token: tokens.refreshToken,
    userId: user._id,
    expiresAt: new Date(Date.now() + 7 * 24 * 60 * 60 * 1000),
  })

  res.json(tokens)
})
```

## Refresh Route

```typescript
app.post('/auth/refresh', async (req, res) => {
  const { refreshToken } = req.body

  try {
    const payload = jwt.verify(refreshToken, JWT_SECRET) as { userId: string }
    const stored = await db.refreshTokens.findOne({ token: refreshToken })

    if (!stored) {
      return res.status(401).json({ error: 'Invalid refresh token' })
    }

    // Delete old, issue new
    await db.refreshTokens.deleteOne({ token: refreshToken })
    const tokens = generateTokens(payload.userId)
    await db.refreshTokens.insertOne({ token: tokens.refreshToken, userId: payload.userId })

    res.json(tokens)
  } catch {
    res.status(401).json({ error: 'Invalid refresh token' })
  }
})
```

## Security Checklist

- Store JWT_SECRET in env vars only
- Short access token lifespan (15 min)
- Rotate refresh tokens on use
- Invalidate refresh tokens on logout
- Use HTTPS only
