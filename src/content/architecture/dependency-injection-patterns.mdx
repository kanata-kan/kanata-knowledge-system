## Constructor Injection

```typescript
// Define interfaces
interface UserRepository {
  findById(id: string): Promise<User | null>
  save(user: User): Promise<void>
}

interface EmailService {
  send(to: string, subject: string, body: string): Promise<void>
}

// Service depends on abstractions
class UserService {
  constructor(
    private readonly userRepo: UserRepository,
    private readonly email: EmailService,
  ) {}

  async createUser(data: CreateUserInput): Promise<User> {
    const user = new User(data)
    await this.userRepo.save(user)
    await this.email.send(user.email, 'Welcome!', 'Thanks for joining')
    return user
  }
}
```

## Factory Pattern

```typescript
function createUserService(deps?: {
  userRepo?: UserRepository
  email?: EmailService
}) {
  const userRepo = deps?.userRepo ?? new MongoUserRepository()
  const email = deps?.email ?? new SmtpEmailService()

  return new UserService(userRepo, email)
}

// Production
const service = createUserService()

// Testing
const service = createUserService({
  userRepo: new InMemoryUserRepository(),
  email: new MockEmailService(),
})
```

## Simple Context Object

```typescript
type AppContext = {
  db: Database
  cache: CacheService
  logger: Logger
}

function createContext(): AppContext {
  return {
    db: new PostgresDatabase(env.DATABASE_URL),
    cache: new RedisCache(env.REDIS_URL),
    logger: new ConsoleLogger(),
  }
}

// Pass context through your app
async function handleRequest(ctx: AppContext, req: Request) {
  const user = await ctx.db.users.findById(req.userId)
  ctx.logger.info(`User ${user.name} made request`)
}
```

## Benefits

- Testable code (swap real dependencies for mocks)
- Explicit dependencies (no hidden globals)
- Flexible composition
- No framework needed
