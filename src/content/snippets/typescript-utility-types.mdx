## Pick & Omit

```typescript
type User = { id: string; name: string; email: string; password: string }

// Only include specific keys
type PublicUser = Pick<User, 'id' | 'name' | 'email'>

// Exclude specific keys
type CreateUserInput = Omit<User, 'id'>
```

## Partial & Required

```typescript
type User = { id: string; name: string; email: string }

// All fields optional
type UpdateUserInput = Partial<User>

// All fields required (reverse of Partial)
type CompleteUser = Required<User>
```

## Record

```typescript
// Map string keys to a specific value type
type StatusMap = Record<string, boolean>

// Constrained keys
type RolePermissions = Record<'admin' | 'user' | 'guest', string[]>
```

## Extract & Exclude

```typescript
type Status = 'active' | 'inactive' | 'pending' | 'deleted'

// Only keep matching members
type ActiveStatuses = Extract<Status, 'active' | 'pending'>
// Result: 'active' | 'pending'

// Remove matching members
type VisibleStatuses = Exclude<Status, 'deleted'>
// Result: 'active' | 'inactive' | 'pending'
```

## ReturnType & Parameters

```typescript
function createUser(name: string, email: string) {
  return { id: '1', name, email }
}

type UserResult = ReturnType<typeof createUser>
// { id: string; name: string; email: string }

type UserParams = Parameters<typeof createUser>
// [name: string, email: string]
```

## Custom Utility: DeepPartial

```typescript
type DeepPartial<T> = {
  [P in keyof T]?: T[P] extends object ? DeepPartial<T[P]> : T[P]
}
```

## Custom Utility: NonNullableFields

```typescript
type NonNullableFields<T> = {
  [P in keyof T]: NonNullable<T[P]>
}
```
