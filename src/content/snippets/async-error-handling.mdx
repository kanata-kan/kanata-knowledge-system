## The Go-Style Tuple Pattern

Avoid nested try-catch blocks:

```typescript
type Result<T> = [Error, null] | [null, T]

async function to<T>(promise: Promise<T>): Promise<Result<T>> {
  try {
    const data = await promise
    return [null, data]
  } catch (error) {
    return [error as Error, null]
  }
}
```

## Usage

```typescript
const [error, user] = await to(getUser(id))

if (error) {
  console.error('Failed to get user:', error.message)
  return
}

// user is typed correctly here
console.log(user.name)
```

## Multiple Async Calls

```typescript
const [userErr, user] = await to(getUser(id))
if (userErr) return handleError(userErr)

const [postsErr, posts] = await to(getUserPosts(user.id))
if (postsErr) return handleError(postsErr)

return { user, posts }
```

## Safe JSON Parse

```typescript
function safeJsonParse<T>(json: string): Result<T> {
  try {
    return [null, JSON.parse(json) as T]
  } catch (error) {
    return [error as Error, null]
  }
}

const [err, data] = safeJsonParse<Config>(rawConfig)
```

## Promise.allSettled for Batch Operations

```typescript
const results = await Promise.allSettled([
  sendEmail(user1),
  sendEmail(user2),
  sendEmail(user3),
])

const failures = results
  .filter((r): r is PromiseRejectedResult => r.status === 'rejected')
  .map(r => r.reason)

if (failures.length > 0) {
  console.error('Some emails failed:', failures)
}
```
