## The Result Type

```typescript
type Result<T, E = Error> =
  | { ok: true; data: T }
  | { ok: false; error: E }

function ok<T>(data: T): Result<T, never> {
  return { ok: true, data }
}

function err<E>(error: E): Result<never, E> {
  return { ok: false, error }
}
```

## Usage

```typescript
async function getUser(id: string): Promise<Result<User, 'NOT_FOUND' | 'DB_ERROR'>> {
  try {
    const user = await db.users.findById(id)
    if (!user) return err('NOT_FOUND')
    return ok(user)
  } catch {
    return err('DB_ERROR')
  }
}

// Consumer
const result = await getUser('123')

if (result.ok) {
  console.log(result.data.name) // typed as User
} else {
  switch (result.error) {
    case 'NOT_FOUND': return res.status(404).json({ error: 'User not found' })
    case 'DB_ERROR': return res.status(500).json({ error: 'Internal error' })
  }
}
```

## Chaining Results

```typescript
function map<T, U, E>(result: Result<T, E>, fn: (data: T) => U): Result<U, E> {
  if (result.ok) return ok(fn(result.data))
  return result
}

function flatMap<T, U, E>(result: Result<T, E>, fn: (data: T) => Result<U, E>): Result<U, E> {
  if (result.ok) return fn(result.data)
  return result
}
```

## Why Use This Pattern?

- Forces callers to handle errors (no silent crashes)
- Errors are typed and exhaustive
- No try/catch pollution
- Composable with map/flatMap
- Perfect for service layer boundaries
