## Basic Template Literals

```typescript
type EventName = `on${Capitalize<'click' | 'focus' | 'blur'>}`
// "onClick" | "onFocus" | "onBlur"

type CSSUnit = `${number}${'px' | 'rem' | 'em' | '%' | 'vh' | 'vw'}`
// "10px" | "1.5rem" | "100%" etc.
```

## Route Type Safety

```typescript
type Route = `/${string}`
type APIRoute = `/api/${string}`

function navigate(path: Route) { /* ... */ }
navigate('/dashboard')  // ✓
navigate('dashboard')   // ✗ Error
```

## Event Handler Map

```typescript
type EventMap = {
  click: { x: number; y: number }
  keydown: { key: string }
  scroll: { delta: number }
}

type HandlerName<T extends string> = `on${Capitalize<T>}`

type EventHandlers = {
  [K in keyof EventMap as HandlerName<K & string>]: (event: EventMap[K]) => void
}

// Result:
// {
//   onClick: (event: { x: number; y: number }) => void
//   onKeydown: (event: { key: string }) => void
//   onScroll: (event: { delta: number }) => void
// }
```

## String Parsing

```typescript
type ExtractParam<T extends string> =
  T extends `${infer _Start}:${infer Param}/${infer Rest}`
    ? Param | ExtractParam<Rest>
    : T extends `${infer _Start}:${infer Param}`
      ? Param
      : never

type Params = ExtractParam<'/users/:id/posts/:postId'>
// "id" | "postId"
```

## Intrinsic String Types

```typescript
Uppercase<'hello'>     // "HELLO"
Lowercase<'HELLO'>     // "hello"
Capitalize<'hello'>    // "Hello"
Uncapitalize<'Hello'>  // "hello"
```
